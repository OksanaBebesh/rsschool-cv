          SQL Query For finding errors with health cards, when cards have two patients
          SELECT DISTINCT 
          t.Health_card,
          t.PatientLastName,
          t.PatientFirstName,
          t.Birthdate  
          FROM patient t
          JOIN
              (SELECT  DISTINCT 
              p.Health_card,
              p.PatientLastName,
              p.PatientFirstName,
              p.Birthdate
              FROM patient p 
              ) as t2
          ON 
          t.Health_card = t2.Health_card  AND
          t.PatientLastName != t2.PatientLastName AND
          t.PatientFirstName != t2.PatientFirstName
          t.Birthdate != t2.Birthdate;
